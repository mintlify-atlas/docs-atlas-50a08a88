---
title: 'Create Supplier'
api: 'POST /api/suppliers'
description: 'Create a new supplier record in the system'
---

## Request Body

<ParamField body="codprovee" type="string" required>
  Unique supplier code identifier (max 50 characters)
</ParamField>

<ParamField body="nombre" type="string" required>
  Supplier name (max 200 characters)
</ParamField>

<ParamField body="telefono" type="string">
  Contact phone number (max 15 characters)
</ParamField>

<ParamField body="email" type="string">
  Supplier email address (max 200 characters)
</ParamField>

<ParamField body="direccion" type="string">
  Physical address (max 300 characters)
</ParamField>

<ParamField body="estado" type="integer">
  Supplier status (1 = active, 0 = inactive). Default: 1
</ParamField>

## Response

<ResponseField name="codprovee" type="string">
  The created supplier's code
</ResponseField>

<ResponseField name="nombre" type="string">
  Supplier name
</ResponseField>

<ResponseField name="telefono" type="string">
  Contact phone number
</ResponseField>

<ResponseField name="email" type="string">
  Supplier email address
</ResponseField>

<ResponseField name="direccion" type="string">
  Physical address
</ResponseField>

<ResponseField name="estado" type="integer">
  Supplier status
</ResponseField>

<ResponseField name="createdDate" type="datetime">
  Creation timestamp
</ResponseField>
